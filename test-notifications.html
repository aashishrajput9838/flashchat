FlashChat Notification Test

Instructions:

  1. Make sure your FlashChat app is running at http://localhost:5173
  2. Open the FlashChat app in your browser and log in
  3. Grant notification permissions when prompted
  4. Open the browser's developer console (F12)
  5. Copy and paste the code below into the console

Test Code:

// Get current user const user = getCurrentUser(); console.log("Current User ID:", user.uid); // Get FCM token from Firestore getDoc(doc(db, "users", user.uid)).then(userDoc => { const userData = userDoc.data(); const fcmToken = userData?.fcmToken; console.log("FCM Token:", fcmToken ? fcmToken.substring(0, 20) + "..." : "Not found"); // Send test notification if (fcmToken) { fetch("http://localhost:3001/api/test-notification", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ token: fcmToken, title: "Test Notification", body: "This is a test notification from FlashChat" }) }).then(response => response.json()) .then(result => { console.log("Notification Test Result:", result); // Display result in the UI const resultDiv = document.getElementById('result'); if (result.success) { resultDiv.innerHTML = '✅ Test notification sent successfully! Check for a notification in your browser.'; resultDiv.className = 'result success'; } else { resultDiv.innerHTML = `❌ Failed to send notification: ${result.error}`; resultDiv.className = 'result error'; } }) .catch(error => { console.error("Error sending notification:", error); const resultDiv = document.getElementById('result'); resultDiv.innerHTML = `❌ Error sending notification: ${error.message}`; resultDiv.className = 'result error'; }); } else { console.log("No FCM token found"); const resultDiv = document.getElementById('result'); resultDiv.innerHTML = '❌ No FCM token found for user. Make sure you have granted notification permissions.'; resultDiv.className = 'result error'; } });
